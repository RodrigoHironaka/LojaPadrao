CREATE TABLE cidade(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(100) NOT NULL,
uf VARCHAR(2) NOT NULL,
`status` CHAR(1),
PRIMARY KEY(id)
);

CREATE TABLE grupo(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(100) NOT NULL,
`status` CHAR(1),
PRIMARY KEY(id)
);

CREATE TABLE grupo(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(100) NOT NULL,
`status` CHAR(1),
PRIMARY KEY(id)
);

CREATE TABLE departamento(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(100) NOT NULL,
`status` CHAR(1),
PRIMARY KEY(id)
);

CREATE TABLE funcao(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
descricao VARCHAR(300),
`status` CHAR(1),
PRIMARY KEY(id)
);

CREATE TABLE usuario(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
senha varchar(15) not null,
confirmaSenha varchar(15) not null,
tipo VARCHAR(20),
`status` CHAR(1),
PRIMARY KEY(id)
);

CREATE TABLE cliente(
id int NOT NULL AUTO_INCREMENT,
nomefantasia VARCHAR(100) NOT NULL,
razaosocial VARCHAR(100),
rgie VARCHAR(20),
cpfcnpj VARCHAR(20),
endereco VARCHAR(100),
numero VARCHAR(10),
complemento VARCHAR(20),
bairro VARCHAR(20),
cep VARCHAR(10),
idCidade int,
email VARCHAR(50),
telefone VARCHAR(13),
celular VARCHAR(14),
celular2 VARCHAR(14),
observacao VARCHAR(300),
dataCadastro Date,
`status` char(1),
PRIMARY KEY(id),
constraint fk_cliente_cidade FOREIGN KEY (idCidade)
  REFERENCES cidade (id)
);

CREATE TABLE fornecedor(
id int NOT NULL AUTO_INCREMENT,
nomeVendedor VARCHAR(100) NOT NULL,
nomeFantasia VARCHAR(100) NOT NULL,
razaoSocial VARCHAR(100),
ie VARCHAR(20),
cnpj VARCHAR(20),
endereco VARCHAR(100),
numero VARCHAR(10),
complemento VARCHAR(20),
bairro VARCHAR(20),
cep VARCHAR(10),
idCidade int,
email VARCHAR(50),
telefone VARCHAR(13),
celular VARCHAR(14),
celular2 VARCHAR(14),
observacao VARCHAR(300),
dataCadastro varchar(50),
`status` char(1),
foto blob,
PRIMARY KEY(id),
constraint fk_fornecedor_cidade FOREIGN KEY (idCidade)
  REFERENCES cidade (id)
);

CREATE TABLE formaPagamento(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
qtdParcelas int,
`status` CHAR(1),
PRIMARY KEY(id)
);

CREATE TABLE unidademedida(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(100) NOT NULL,
sigla VARCHAR(11) NOT NULL,
`status` char(1),
PRIMARY KEY (id)
);

CREATE TABLE marca(
id int NOT NULL AUTO_INCREMENT,
nome VARCHAR(50) NOT NULL,
modelo VARCHAR(50),
ano int,
`status` char(1),
PRIMARY KEY (id)
);

CREATE TABLE produto(
id int NOT NULL AUTO_INCREMENT,
numSerie int,
tipoProduto VARCHAR(20),
dataCadastro VARCHAR(30),
nome VARCHAR(100) NOT NULL,
apelido VARCHAR(100),
idUN int,
idGrupo int,
idSubGrupo int,
idFornecedor int,
idMarca int,
precoCusto DECIMAL,
porcentagemCusto DECIMAL,
precoAvista DECIMAL,
porcentagemAvista DECIMAL,
precoPrazo DECIMAL,
porcentagemDesconto DECIMAL,
precoDesconto DECIMAL,
estoqueAtual DECIMAL,
estoqueMax DECIMAL,
estoqueMin DECIMAL,
controleEstoque char(1),
observacao VARCHAR(300),
foto LONGBLOB,
`status` char(1),
PRIMARY KEY(id),
constraint fk_produto_UN FOREIGN KEY (idUN)
  REFERENCES unidademedida (id),
constraint fk_produto_Grupo FOREIGN KEY (idGrupo)
  REFERENCES grupo (id),
constraint fk_produto_SubGrupo FOREIGN KEY (idSubGrupo)
  REFERENCES subgrupo (id),
constraint fk_produto_Fornecedor FOREIGN KEY (idFornecedor)
  REFERENCES fornecedor (id),
constraint fk_produto_Marca FOREIGN KEY (idMarca)
  REFERENCES marca (id)
);

create table produtoImposto(
id int not null AUTO_INCREMENT,
idProduto int not null,
icms int,
cfop int,
ncm int,
cest int,
eanGtin int,
PRIMARY KEY(id),
CONSTRAINT fk_produto_imposto FOREIGN KEY (idProduto)
  REFERENCES produto (id)
);
